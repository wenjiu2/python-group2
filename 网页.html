from pptx import Presentation
from pptx.util import Inches
import smtplib,email
chst = email.charset.Charset(input_charset = 'utf-8')

# header里分别定义发件人,收件人以及邮件主题。
header = ("From: %s\nTo: %s\nSubject: %s\n\n" %
("sendmail@163.com",
"receivemail@163.com",
chst.header_encode("Mail Test")))
msg = ''' '''

# 1. 得到幻灯片对象
prs = Presentation("第4章 条件、循环和其他语句.pptx")
# 2. 获取prs对象中的每一页PPT
for slide in prs.slides:
# 3. 遍历获取每页PPT中的内容
for shape in slide.shapes:
# 4. 判断是否存在文本
if shape.has_text_frame:
# 5. 获取shape中的文本
text_frame = shape.text_frame
# 6. 获取text_frame中的段落内容
for paragraph in text_frame.paragraphs:
# 7. 打印段落内容
msg += paragraph.text.strip()+'\n'
# 对header和msg邮件正文进行utf-8编码，指定发信人的smtp服务器，并输入邮箱密码进行登录验证，最后发送邮件。
email_con = header.encode('utf-8') + msg.encode('utf-8')
smtp = smtplib.SMTP("smtp.163.com")
smtp.login("发送人邮箱地址@163.com","PGHBUFOPJLRGJVBA")
smtp.sendmail('发送人邮箱地址.com','接收人邮件地址@qq.com',email_con)
smtp.quit()
